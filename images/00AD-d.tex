
\begin{tikzpicture}
  \pgfmathsetmacro\yA{2}
  \pgfmathsetmacro\xC{6}
  \pgfmathsetmacro\rad{24/sqrt(\yA^2+\xC^2)}
  \tkzDefPoints{0/\yA/A,0/0/B,\xC/0/C}
  \tkzDefPointBy[rotation=center A angle 60](C)
  \tkzInterLC[R](A,tkzPointResult)(A,\rad)
  \tkzGetSecondPoint{D}
  \tkzDefLine[altitude](A,D,B) \tkzGetPoint{H}
  \tkzDrawPolygon(A,B,C,D)
  \tkzDrawSegments(A,C B,D)
  \tkzDrawSegments[dashed](A,H D,H)
  \tkzLabelAngle[pos=.5](C,A,D){$60^\circ$}
  \tkzLabelAngle[pos=.4](B,A,C){$\alpha$}
  \tkzMarkRightAngles(C,B,A B,H,D)
  \tkzDrawPoints(A,B,C,D,H)
  \tkzLabelPoints[left](A,B,H)
  \tkzLabelPoints[right](C)
  \tkzLabelPoints[above](D)
